let S = null;
let saveTimer = null;
let savePending = false;

function showSync(status, msg) {
  const el = document.getElementById('sync-status');
  el.className = status;
  el.textContent = msg;
  if (status === 'ok') setTimeout(() => el.classList.add('hidden'), 2000);
}

async function cargarDesdeServidor() {
  try {
    const res = await fetch('/api/state');
    const data = await res.json();
    if (data.ok && data.state) {
      S = data.state;
      showSync('ok', '✓ Datos cargados');
    } else {
      S = defS();
    }
  } catch (e) {
    console.warn('Sin conexión al servidor, usando localStorage');
    const local = localStorage.getItem('inv_v9');
    S = local ? JSON.parse(local) : defS();
  }
  render();
}

async function guardarEnServidor() {
  savePending = false;
  showSync('saving', '⏳ Guardando...');
  try {
    localStorage.setItem('inv_v9', JSON.stringify(S)); // backup local
    const res = await fetch('/api/state', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ state: S })
    });
    const data = await res.json();
    if (data.ok) showSync('ok', '✓ Guardado en Sheets');
    else showSync('error', '⚠ Error al guardar');
  } catch (e) {
    showSync('error', '⚠ Sin conexión — guardado local');
  }
}

async function save() {
  localStorage.setItem('inv_v9', JSON.stringify(S));
  if (saveTimer) clearTimeout(saveTimer);
  saveTimer = setTimeout(guardarEnServidor, 1500); // debounce 1.5s
}

function hoy() { return new Date().toISOString().split('T')[0]; }
function man() { return fmas(hoy(), 1); }
function fmas(iso, n) { const d = new Date(iso); d.setDate(d.getDate()+n); return d.toISOString().split('T')[0]; }
function fd(iso) { if (!iso) return '—'; const [y,m,d] = iso.split('-'); return `${d}/${m}/${y}`; }
function dd(iso) { return Math.floor((new Date() - new Date(iso)) / 86400000); }

function delLote(id) {
  const l=S.lotes.find(x=>x.id===id);

  S.lotes=S.lotes.filter(x=>x.id!==id);
  log('Eliminación',`Lote ${l?l.varNom:'?'} eliminado`); save(); cm('modal-lote'); cm('modal-banc'); render();
}

  movCtx={loteId:id};
  const l=S.lotes.find(x=>x.id===id); if(!l) return;
  const sig={plantines:'engorda',engorda:'adulto'}[l.etapa]; if(!sig) return;
  const tipo=sig==='engorda'?'eng':'adu';
  const maxBanc=sig==='engorda'?8:16;
  const maxP=sig==='engorda'?20*PT:10*PT;
  const obligatorio=l.etapa==='plantines';
  el('mm-t').textContent=`${l.varNom}: ${l.etapa} → ${sig}`;
  let h=`<div class="mc" style="margin-bottom:8px;font-size:11px">Plantas disponibles: <strong>${l.plantasRestantes}</strong> (${fracTubosStr(l.plantasRestantes)} tubos)</div>
  <label>Fecha del movimiento</label><input type="date" id="mov-fecha" value="${hoy()}">
  <label>Bancal destino ${obligatorio?'<span style="color:var(--text-danger)">* obligatorio</span>':''}</label>
  <select id="mov-banc" onchange="checkCap()"><option value="">-- Selecciona bancal --</option>`;
  for (let i=1;i<=maxBanc;i++) {
    const k=`${tipo}_${i}`; const usP=plantasEnBanc(k); const libP=maxP-usP;
    const slots=getBanc(k); const sl=slots.length?slots.map(x=>x.varNom+'×'+x.plantas+'pl').join(', '):'vacío';
    h+=`<option value="${k}" data-libre="${libP}"${libP<=0?' disabled':''}>${sig==='engorda'?'E':'A'}${i} — ${fracTubosStr(libP)} tubos libres (${sl})</option>`;
  }
  h+=`</select>
  ${obligatorio?'<div id="mov-banc-err" style="font-size:10px;color:var(--text-danger);display:none;margin-top:3px">Debes seleccionar un bancal.</div>':''}
  <label>Plantas a mover (máx. ${l.plantasRestantes})</label>
  <input type="number" id="mov-plantas" value="${l.plantasRestantes}" min="1" max="${l.plantasRestantes}" oninput="checkCap()">
  <div id="mov-info" style="font-size:10px;color:var(--text-muted);margin-top:2px">${fracTubosStr(l.plantasRestantes)} tubos equivalentes</div>

function checkCap() {
  const sel=el('mov-banc'),ti=el('mov-plantas'); if(!sel||!ti) return;
  const p=parseInt(ti.value)||0;
  const inf=el('mov-info'); if(inf) inf.textContent=`= ${fracTubosStr(p)} tubos equivalentes`;
  const opt=sel.options[sel.selectedIndex];
  const libreP=opt?parseInt(opt.getAttribute('data-libre')):null;
  const al=el('mov-cap-al');
  if (sel.value&&libreP!==null&&p>libreP) al.innerHTML=`<div class="al r" style="margin-top:4px"><i class="ti ti-alert-triangle"></i>Solo ${libreP} plantas caben en ese bancal (${fracTubosStr(libreP)} tubos).</div>`;
  else al.innerHTML='';
}

  const l=S.lotes.find(x=>x.id===movCtx.loteId); if(!l){cm('modal-mov');return;}

  const bancKey=el('mov-banc').value;
  const obligatorio=l.etapa==='plantines';
  if (obligatorio&&!bancKey) { const e=el('mov-banc-err'); if(e) e.style.display='block'; el('mov-banc').style.borderColor='var(--border-danger)'; return; }
  const fechaMov=el('mov-fecha').value||hoy();
  const plantasM=Math.min(parseInt(el('mov-plantas').value)||l.plantasRestantes,l.plantasRestantes);
  const nota=el('mov-nota').value.trim();
  if (bancKey) { const maxP=sig==='engorda'?20*PT:10*PT; const libP=maxP-plantasEnBanc(bancKey); if(plantasM>libP){alert(`Solo hay espacio para ${libP} plantas en ese bancal.`);return;} }

    el('modal-merma').classList.add('open');
  } else execMov(null);
}

function resMerma(tipo) { cm('modal-merma'); execMov(tipo); }

function execMov(mermaRes) {
  const l=S.lotes.find(x=>x.id===movCtx.loteId); if(!l) return;
  const {sig,fechaMov,bancKey,plantasM,nota,restantes,etapaAnt}=movCtx;
  if (bancKey) addSlot(bancKey,l.varId,l.varNom,plantasM);
  if (!l.movimientos) l.movimientos=[];
  const bL=bancKey?` → Bancal ${bancKey}`:'';
  const mT=restantes>0&&mermaRes==='merma'?` | ${restantes} plantas merma`:'';
  const pT=restantes>0&&mermaRes==='pendiente'?` | ${restantes} plantas pendientes`:'';
  l.movimientos.push({fecha:fechaMov,accion:`→ ${sig}`,detalle:`${plantasM} plantas (${fracTubosStr(plantasM)} tubos)${bL}${nota?' · '+nota:''}${mT}${pT}`});
  if (mermaRes==='merma'&&restantes>0) {
    if (!S.merma) S.merma={plantines:0,engorda:0,adulto:0};
    S.merma[etapaAnt]=(S.merma[etapaAnt]||0)+restantes;
    l.movimientos.push({fecha:fechaMov,accion:'Merma',detalle:`${restantes} plantas (${fracTubosStr(restantes)} tubos) en ${etapaAnt}`});
    log('Merma',`${l.varNom} en ${etapaAnt}: ${restantes} plantas`);
  }
  if (mermaRes==='pendiente'&&restantes>0) {
    const nl={...l,id:S.nextId++,plantas:plantasM,plantasRestantes:plantasM,etapa:sig,fechaEtapa:fechaMov,
      bancalId:bancKey||null,movimientos:[{fecha:fechaMov,accion:`→ ${sig}`,detalle:`${plantasM} plantas${bL} (separado)`}]};
    if (sig==='adulto') nl.fechaVenta=fmas(fechaMov,l.da);
    S.lotes.push(nl); l.plantasRestantes=restantes;
  } else {
    l.etapa=sig; l.fechaEtapa=fechaMov; l.plantasRestantes=plantasM;
    l.bancalId=bancKey||l.bancalId;

function getVentaData() {
  const lotes=S.lotes||[], plan=S.plan||[], vars=(S.vars||[]).map(v=>v.nombre);
  const REAL=[], PROY=[];
  lotes.filter(l=>l.etapa!=='cosechado'&&l.plantasRestantes>0).forEach(l=>REAL.push({fecha:l.fechaVenta,var:l.varNom,plantas:l.plantasRestantes}));
  const hoyD=new Date();
  plan.forEach(p=>{
    let ultima=p.ultimaSiembra?new Date(p.ultimaSiembra):new Date(hoyD.getTime()-p.freq*86400000);
    for (let i=0;i<8;i++) {
      ultima=new Date(ultima.getTime()+p.freq*86400000);
      if (ultima>new Date(hoyD.getTime()+150*86400000)) break;
      PROY.push({fecha:new Date(ultima.getTime()+(p.dp+p.de+p.da)*86400000).toISOString().split('T')[0],var:p.varNom,plantas:p.plantas});
    }
  });
  return {REAL,PROY,vars};
}
